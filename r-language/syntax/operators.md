# Toán Tử

```
# + : phép cộng 2 vector
v <- c( 2,5.5,6)
t <- c(8, 3, 4)
print(v+t)

# kết quả
[1] 10.0  8.5  10.0
```

```
# - : phép trừ 2 Vector
v <- c( 2,5.5,6)
t <- c(8, 3, 4)
print(v-t)

# kết quả
[1] -6.0  2.5  2.0
```

```
# * : phép nhân 2 vector
v <- c( 2,5.5,6)
t <- c(8, 3, 4)
print(v*t)

# kết quả
[1] 16.0 16.5 24.0
```

```
# / : phép chia 2 vector
v <- c( 2,5.5,6)
t <- c(8, 3, 4)
print(v/t)

# kết quả
[1] 0.250000 1.833333 1.500000
```

```
# %% :  Chia vector lấy dư
v <- c( 2,5.5,6)
t <- c(8, 3, 4)
print(v%%t)

# kết quả
[1] 2.0 2.5 2.0
```

```
# %/% :  Chia vector lấy nguyên
v <- c( 2,5.5,6)
t <- c(8, 3, 4)
print(v%/%t)

# kết quả
[1] 0 1 1
```

```
# ^ :  phép lũy thừa
v <- c( 2,5.5,6)
t <- c(8, 3, 4)
print(v^t)

# kết quả
[1]  256.000  166.375 1296.000
```
